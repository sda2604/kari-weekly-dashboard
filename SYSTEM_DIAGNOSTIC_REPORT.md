import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from email.mime.base import MIMEBase
from email import encoders
from dotenv import load_dotenv
import os

load_dotenv()

def send_weekly_report(html_file, subject="–ï–∂–µ–Ω–µ–¥–µ–ª—å–Ω—ã–π –æ—Ç—á—ë—Ç KARI"):
    sender = os.getenv('EMAIL_ADDRESS')
    password = os.getenv('EMAIL_PASSWORD')
    recipients = os.getenv('RECIPIENTS').split(',')

    msg = MIMEMultipart()
    msg['From'] = sender
    msg['To'] = ', '.join(recipients)
    msg['Subject'] = subject

    # –¢–µ–∫—Å—Ç –ø–∏—Å—å–º–∞
    body = """
    –î–æ–±—Ä—ã–π –¥–µ–Ω—å!

    –í–æ –≤–ª–æ–∂–µ–Ω–∏–∏ –µ–∂–µ–Ω–µ–¥–µ–ª—å–Ω—ã–π –∞–Ω–∞–ª–∏—Ç–∏—á–µ—Å–∫–∏–π –¥–∞—à–±–æ—Ä–¥ KARI.

    –ö–ª—é—á–µ–≤—ã–µ –ø–æ–∫–∞–∑–∞—Ç–µ–ª–∏ –∑–∞ –Ω–µ–¥–µ–ª—é:
    - –ü—Ä–æ–¥–∞–∂–∏: 15.2 –º–ª–Ω ‚ÇΩ (‚Üë12% –∫ –ø—Ä–æ—à–ª–æ–π –Ω–µ–¥–µ–ª–µ)
    - KOP –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç: 3.45 (–Ω–æ—Ä–º–∞)
    - –ö—Ä–∏—Ç–∏—á–µ—Å–∫–∏–µ –ø—Ä–æ–±–ª–µ–º—ã: 2 –º–∞–≥–∞–∑–∏–Ω–∞ —Å –Ω–µ—Ö–≤–∞—Ç–∫–æ–π —Ç–æ–≤–∞—Ä–∞

    –ü–æ–¥—Ä–æ–±–Ω–æ—Å—Ç–∏ –≤ –¥–∞—à–±–æ—Ä–¥–µ.

    –° —É–≤–∞–∂–µ–Ω–∏–µ–º,
    –°–∏—Å—Ç–µ–º–∞ –∞–Ω–∞–ª–∏—Ç–∏–∫–∏ KARI
    """
    msg.attach(MIMEText(body, 'plain'))

    # –ü—Ä–∏–∫—Ä–µ–ø–∏—Ç—å HTML –¥–∞—à–±–æ—Ä–¥
    with open(html_file, 'rb') as f:
        part = MIMEBase('application', 'octet-stream')
        part.set_payload(f.read())
        encoders.encode_base64(part)
        part.add_header('Content-Disposition', f'attachment; filename={os.path.basename(html_file)}')
        msg.attach(part)

    # –û—Ç–ø—Ä–∞–≤–∫–∞
    with smtplib.SMTP('smtp.gmail.com', 587) as server:
        server.starttls()
        server.login(sender, password)
        server.send_message(msg)

    print(f"‚úì –û—Ç—á—ë—Ç –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω: {', '.join(recipients)}")

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
send_weekly_report('output/dashboard_final.html')
